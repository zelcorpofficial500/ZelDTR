<script>
// Live clock function to give employees immediate visual verification of logging time
function updateClock() {
const now = new Date();
let hours = now.getHours();
const minutes = String(now.getMinutes()).padStart(2, '0');
const seconds = String(now.getSeconds()).padStart(2, '0');
const ampm = hours >= 12 ? 'PM' : 'AM';
hours = hours % 12;
hours = hours ? hours : 12; // structural conversion for hour formats
document.getElementById('clockDisplay').innerText = `${String(hours).padStart(2, '0')}:${minutes}:${seconds} ${ampm}`;
}
setInterval(updateClock, 1000);
updateClock();

// Standard PWA Service worker installation registry
if ('serviceWorker' in navigator) {
navigator.serviceWorker.register('sw.js').then(() => console.log('DTR Pipeline Ready'));
}
